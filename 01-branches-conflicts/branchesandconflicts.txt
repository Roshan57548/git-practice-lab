# Start on main branch
$ git branch

# Create new branch called feature/profile-title
$ git branch feature/profile-title

# Switch to the new branch
$ git checkout feature/profile-title

# To create and switch in one command (older way):
git checkout -b feature/profile-title

# Force delete unmerged branch (local)
git branch -D feature/profile-title

# Delete remote
git push origin --delete feature/profile-title
